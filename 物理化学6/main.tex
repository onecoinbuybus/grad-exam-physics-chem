\[

\importantbox{

  \[ E = nh\nu \quad (n = 1, 2, 3, \ldots) \]

$h = \SI{6.626e-34}{J.s}$
\end{column}

\section{Schrödinger方程式}
% %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

% ============================================================
\begin{frame}{Schrödinger 方程式の導出}

平面波 $\Psi(x,t) = A\exp\!\bigl[2\pi i(x/\lambda - \nu t)\bigr]$ に
$\lambda = 2\pi\hbar/p$，$E = h\nu$ を代入：

\importantbox{
  $\displaystyle \Psi(x,t) = A \exp\!\left[\frac{i}{\hbar}(px - Et)\right]$
}

偏微分すると演算子が得られる：
\begin{align}
  \hat{p} &= -i\hbar\frac{\partial}{\partial x}
    & &\text{（運動量演算子）} \\[6pt]
  \hat{E} &= i\hbar\frac{\partial}{\partial t}
    & &\text{（エネルギー演算子）}
\end{align}

古典的エネルギー $E = p^2/(2m) + V(x)$ に代入：

\importantbox{
  \textbf{時間依存 Schrödinger 方程式：}\quad
  $\displaystyle \left[-\frac{\hbar^2}{2m}\frac{\partial^2}{\partial x^2} + V\right]\Psi
  = i\hbar\frac{\partial}{\partial t}\Psi$
}
\end{frame}

% ============================================================
\begin{frame}{ハミルトニアンと三次元拡張}

コンパクトに書くと：
\[
  \hat{H}\Psi = i\hbar\frac{\partial}{\partial t}\Psi, \qquad
  \hat{H} = -\frac{\hbar^2}{2m}\nabla^2 + V \quad \text{（ハミルトニアン）}
\]

\vspace{0.5em}
\textbf{変数分離} $\Psi(x,t) = \psi(x)\cdot T(t)$ を代入して整理：
\[
  \underbrace{\frac{1}{\psi}\!\left(-\frac{\hbar^2}{2m}\frac{d^2\psi}{dx^2} + V\psi\right)}_{x\text{のみ}}
  = \underbrace{\frac{i\hbar}{T}\frac{dT}{dt}}_{t\text{のみ}}
  = E
\]

\importantbox{
  \textbf{時間に依存しないSchrödinger方程式（固有値方程式）：}
  \[
    \hat{H}\psi(x) = E\,\psi(x) \quad \Longleftrightarrow \quad
    -\frac{\hbar^2}{2m}\frac{d^2\psi}{dx^2} + V(x)\psi = E\psi
  \]
}

時間部分：$T(t) = e^{-iEt/\hbar}$ → 全体 $\Psi(x,t) = \psi(x)\,e^{-iEt/\hbar}$
\end{frame}

\section{数学基礎：変数分離法}
% %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

% ============================================================
\begin{frame}{数学基礎：なぜ先に波動方程式を学ぶのか}

Schrödinger方程式は二階偏微分方程式の一種。\\
解法の本質を理解するため，まず\textbf{波動方程式}（弦の振動）で練習する。

\[
  \frac{\partial^2 u}{\partial t^2} = c^2 \frac{\partial^2 u}{\partial x^2}
\]

\textbf{\color{accent}解法の流れ（Schrödinger方程式も全く同じ！）：}
\begin{enumerate}
  \item 変数分離 $u(x,t) = f(x)\cdot g(t)$ と置く
  \item 二つの常微分方程式に分離：$f''/f = g''/(c^2g) = \lambda$
  \item $\lambda$の符号を場合分け → $\lambda < 0$のみ物理的に正しい解
  \item 境界条件を適用 → \alert{固有値が離散的に決まる}
\end{enumerate}
\end{frame}

% ============================================================
\begin{frame}{変数分離法の要点}

$f''/f = \lambda$ の解は$\lambda$の符号で決まる：

\vspace{0.5em}
\begin{description}
  \item[\color{alertred}$\lambda > 0$] $f = Ce^{\sqrt{\lambda}\,x} + De^{-\sqrt{\lambda}\,x}$ → 発散，\textbf{不適}
  \item[\color{alertred}$\lambda = 0$] $f = Cx + D$ → 境界条件で自明解のみ，\textbf{不適}
  \item[\color{darkblue}$\lambda < 0$] $f = C\cos(\sqrt{|\lambda|}\,x) + D\sin(\sqrt{|\lambda|}\,x)$ → \textbf{正しい解}
\end{description}

\vspace{0.5em}
境界条件 $f(0) = 0$, $f(L) = 0$ を適用すると：
\[
  f(0) = C = 0, \quad f(L) = D\sin(\sqrt{|\lambda|}L) = 0
\]
$D \neq 0$なので $\sqrt{|\lambda|}L = n\pi$ → \alert{固有値が離散的に決まる}：
\[
  \boxed{f_n(x) = D\sin\!\left(\frac{n\pi x}{L}\right), \quad n = 1, 2, 3, \ldots}
\]
\end{frame}

\section{井戸型ポテンシャル（Particle in a Box）}
% %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

% ============================================================
\begin{frame}{井戸型ポテンシャルの設定}

\begin{columns}[c]
\begin{column}{0.45\textwidth}
  \begin{tikzpicture}[scale=0.85]
    \fill[pattern=north east lines, pattern color=gray] (-0.5,0) rectangle (0,3);
    \fill[pattern=north east lines, pattern color=gray] (4,0) rectangle (4.5,3);
    \draw[very thick] (0,3) -- (0,0) -- (4,0) -- (4,3);
    \fill[yellow!30, opacity=0.5] (0,0) rectangle (4,3);
    \node[below] at (0,-0.2) {$0$};
    \node[below] at (4,-0.2) {$a$};
    \node at (2,1.5) {$V = 0$};
    \node at (-1,1.5) {$V = \infty$};
    \node at (5,1.5) {$V = \infty$};
    \draw[->] (-1.2,0) -- (5.2,0) node[right] {$x$};
    \draw[->] (0,-0.5) -- (0,3.5) node[above] {$V(x)$};
  \end{tikzpicture}
\end{column}
\begin{column}{0.55\textwidth}
  \[
    V(x) = \begin{cases} 0 & (0 \leq x \leq a) \\ \infty & (\text{それ以外}) \end{cases}
  \]

  壁の外：$\psi(x) = 0$（存在確率ゼロ）

  \vspace{0.5em}
  井戸内で $V = 0$ を代入：
  \[
    -\frac{\hbar^2}{2m}\frac{d^2\psi}{dx^2} = E\psi
  \]
  \[
    \frac{d^2\psi}{dx^2} = -k^2\psi, \quad k \equiv \sqrt{\frac{2mE}{\hbar^2}}
  \]
\end{column}
\end{columns}

\vspace{0.3em}
{\small\color{accent}数学基礎の $f'' = \lambda f$（$\lambda = -k^2 < 0$）と全く同じ形！}
\end{frame}

% ============================================================
\begin{frame}{境界条件の適用と量子化}

一般解：$\psi(x) = A\cos(kx) + B\sin(kx)$

\vspace{0.5em}
\textbf{境界条件：}
\begin{align}
  \psi(0) &= A = 0 \quad \Longrightarrow \quad \psi(x) = B\sin(kx) \\[6pt]
  \psi(a) &= B\sin(ka) = 0
\end{align}

$B \neq 0$（さもなくば粒子不存在）なので：
\[
  \sin(ka) = 0 \quad \Longrightarrow \quad ka = n\pi \quad (n = 1, 2, 3, \ldots)
\]

\importantbox{
  \[
    k = \frac{n\pi}{a}, \qquad
    \psi_n(x) = B\sin\!\left(\frac{n\pi x}{a}\right), \quad n = 1, 2, 3, \ldots
  \]
}

{\small ※ $n = 0$ は$\psi = 0$（粒子不存在）なので不適。}

\vspace{0.3em}
{\small\color{accent}【注】$-L \leq x \leq L$モデル（$\cos/\sin$交替）と本質的に等価（座標原点の移動）。\\
入試では$0 \leq x \leq a$モデルが最も多い。}
\end{frame}

% ============================================================
\begin{frame}{規格化}

確率の総和$= 1$の条件：
\[
  \int_0^a |\psi_n(x)|^2\,dx = B^2 \int_0^a \sin^2\!\left(\frac{n\pi x}{a}\right)dx = B^2 \cdot \frac{a}{2} = 1
\]
\[
  \therefore \quad B = \sqrt{\frac{2}{a}}
\]

\importantbox{
  \textbf{規格化された波動関数：}
  \[
    \boxed{\psi_n(x) = \sqrt{\frac{2}{a}}\sin\!\left(\frac{n\pi x}{a}\right), \quad n = 1, 2, 3, \ldots}
  \]
}

\exammark{2025年度 第2問(5)で$\psi = f(\xi)\exp(-\xi^2/2)$の規格化が出題}
\end{frame}

% ============================================================
\begin{frame}{エネルギー固有値}

$k = \sqrt{2mE/\hbar^2} = n\pi/a$ から：

\importantbox{
  \[
    \boxed{E_n = \frac{n^2\pi^2\hbar^2}{2ma^2} = \frac{n^2 h^2}{8ma^2}, \qquad n = 1, 2, 3, \ldots}
  \]
}

\vspace{0.3em}
\textbf{\color{alertred}重要な結論：}
\begin{enumerate}
  \item エネルギーは\alert{離散的}（$n$の整数値のみ）
  \item $n = 0$は不適（$\psi = 0$，粒子不存在）
  \item $E_n \propto n^2/m/a^2$ → 軽い粒子・狭い箱ほど量子効果が大きい
\end{enumerate}

\vspace{0.3em}
\textbf{数値例：}電子（$m = \SI{9.11e-31}{kg}$）を$a = \SI{1}{nm}$の箱に閉じ込めると
\[
  E_1 = \frac{(\SI{6.626e-34}{})^2}{8 \times \SI{9.11e-31}{} \times (\SI{1e-9}{})^2}
  \approx \SI{0.376}{eV} \gg k_\mathrm{B}T \approx \SI{0.025}{eV}
\]
→ 室温でも\alert{量子効果は無視できない}！
\end{frame}

\begin{frame}{零点エネルギー}

基底状態（$n = 1$）：
\[
  E_1 = \frac{h^2}{8ma^2} \neq 0
\]

\begin{center}
\begin{tabular}{lcc}
  \toprule
  & 古典力学 & 量子力学 \\
  \midrule
  最低エネルギー & $E = 0$（静止可能） & $E_1 > 0$（\alert{静止不可能}） \\
  \bottomrule
\end{tabular}
\end{center}

\vspace{0.5em}
\textbf{\color{accent}物理的理由：Heisenbergの不確定性原理}
\[
  \Delta x \cdot \Delta p \geq \frac{\hbar}{2}
\]
箱に閉じ込め：$\Delta x \leq a$ → $\Delta p \geq \hbar/(2a)$\\
→ 運動量がゼロにならない → エネルギーもゼロにならない

\vspace{0.3em}
\exammark{頻出：「零点エネルギーが存在する理由を説明せよ」}
\end{frame}

% ============================================================
\begin{frame}{対応原理と古典極限}

\textbf{対応原理（Bohr）：}
$n \to \infty$のとき，量子力学は古典力学に一致する。

\vspace{0.5em}
\textbf{確認：}
\begin{itemize}
  \item $n = 1$：中央に存在しやすい → 古典と異なる
  \item $n \to \infty$：$|\psi_n|^2$が激しく振動 → 平均で\textbf{一様分布}（古典的！）
\end{itemize}

\vspace{0.3em}
\textbf{エネルギー間隔の相対値：}
\[
  \frac{\Delta E}{E_n} = \frac{2n+1}{n^2} \xrightarrow{n \to \infty} 0
  \quad \text{（事実上連続）}
\]

\textbf{マクロな物体}（$m$大，$a$大）：$E_1 \approx 0$\\
→ エネルギー準位が連続 → 古典力学と一致
\end{frame}

% ============================================================
\begin{frame}{応用：共役ポリエンの$\pi$電子モデル}

共役ポリエンの$\pi$電子を「\textbf{1次元の箱の中の粒子}」で近似。

\vspace{0.3em}
\textbf{\color{accent}ブタジエン（$\text{CH}_2\!=\!\text{CH}\!-\!\text{CH}\!=\!\text{CH}_2$）の例：}
\begin{itemize}
  \item 4個の$\pi$電子，結合数$\approx 3$，結合長$\approx \SI{0.14}{nm}$
  \item $a \approx 3 \times \SI{0.14}{nm} = \SI{0.42}{nm}$
  \item Pauliの排他原理：$n = 1$に2個，$n = 2$に2個 → HOMO: $n = 2$
\end{itemize}

HOMO→LUMO 遷移（$n = 2 \to 3$）：
\[
  \Delta E = (2 \times 2 + 1)\frac{h^2}{8ma^2}
  = \frac{5 \times (\SI{6.626e-34}{})^2}{8 \times \SI{9.11e-31}{} \times (\SI{4.2e-10}{})^2}
  \approx \SI{1.7e-18}{J}
\]
\[
  \lambda = \frac{hc}{\Delta E} \approx \SI{117}{nm}
  \quad \text{（実測$\approx \SI{217}{nm}$，モデルの限界あり）}
\]

\importantbox{共役が長い → $a$増大 → $\Delta E$減少 → $\lambda$が\alert{赤方偏移}}

\exammark{2021年度 第2問 II で共役ポリエンのparticle-in-a-boxが出題}
\end{frame}
